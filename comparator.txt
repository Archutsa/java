package learning;

import java.util.Comparator;

public class CompareDemo implements Comparator {
	
@Override
	public int compare(Object o1, Object o2) {
		
		Students l1=(Students)o1;
		Students l2=(Students)o2;
		int result=l1.name.compareTo(l2.name);
		if(result>0)
			return +1;
		else if(result<0)
			return -1;
		else
			return 0;
	}
}
